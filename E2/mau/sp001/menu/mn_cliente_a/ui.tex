\begin{IU}{MN01A}{Menú para Administrar la Cuenta del Cliente}{Este menú tiene como principal objetivo mostrar las opciones a las que el \getElementById[Actor]{Cliente} tiene acceso en la aplicación móvil para administrar la información de su cuenta como lo es foto de perfil, contraseña, agregar o eliminar una forma de pago, o cerrar sesión en la aplicación.}{menu/mn_cliente_a/MN01A}
	\item[Acciones:]\hspace{1pt}
	\begin{Citemize}
		\item \userAccountIcon{} Al presionar este botón en donde se despliega el nombre del Cliente se redirigirá a la pantalla \getElementById[IU]{IUMC2.4} en donde el Cliente podrá editar la información de su cuenta como por ejemplo la contraseña con la que se ingresará a la aplicación.
		\item \bellIcon{} Al presionar este botón se redirigirá a la pantalla \getElementById[IU]{IUMC2.2}.
		\item \efectivoIcon{Forma de Pago} Al presionar este botón se redirigirá a la pantalla \getElementById[IU]{IUMP2} en donde el cliente podrá agregar o eliminar las formas de pago que tiene disponibles para realizar el pago de una orden.
		\item \IUbutton{Cerrar Sesión} Al presionar este botón se dará por concluida la sesión activa del Cliente y se redirigirá a la pantalla \getElementById[IU]{IUMAU1}
	\end{Citemize}
\end{IU}
